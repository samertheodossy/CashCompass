// Sentinel option appended to the Existing Payee dropdown so users can // explicitly say "I'm adding a brand-new payee not in this list." // Without it the only way to add a new payee was to ignore the visible // dropdown and just type in the second box, which is unclear UX. The // sentinel value is internal only — currentPaymentPayee + the save // path skip it, so the server only ever receives the typed payee text. // Mirrors the BANK_ADD_POLICY_OTHER_SENTINEL_ / INV_ADD_TYPE_OTHER_SENTINEL_ // pattern used elsewhere in this file family. var QUICK_ADD_PAYEE_OTHER_SENTINEL_ = '__OTHER__'; function currentPaymentPayee() { const typed = document.getElementById('pay_payeeInput').value.trim(); if (typed) return typed; const sel = String(document.getElementById('pay_payeeSelect').value || ''); if (sel === QUICK_ADD_PAYEE_OTHER_SENTINEL_) return ''; return sel; } function syncPaymentPayeeInput() { const selected = document.getElementById('pay_payeeSelect').value || ''; const input = document.getElementById('pay_payeeInput'); if (!input) return; if (selected === QUICK_ADD_PAYEE_OTHER_SENTINEL_) { // User explicitly chose "Other (type new payee)" — clear the // typed input and focus it so they can immediately type the new // name. Don't copy the literal sentinel string into the input. input.value = ''; try { input.focus(); } catch (_e) {} return; } input.value = selected; } function applyQuickPaymentPriorMonth_(data) { const labelEl = document.getElementById('pay_priorMonthLabel'); const valEl = document.getElementById('pay_priorMonthPayment'); if (!valEl) return; if (labelEl) labelEl.textContent = data && data.priorMonthLabel ? data.priorMonthLabel : '—'; if (!data) { valEl.textContent = '—'; return; } if (data.priorMonthUnavailableMessage) { valEl.textContent = data.priorMonthUnavailableMessage; return; } if (data.priorMonthValue !== null && data.priorMonthValue !== undefined && data.priorMonthValue !== '') { valEl.textContent = fmtCurrency(data.priorMonthValue); } else { valEl.textContent = '—'; } } /** * Hidden prefill-only state: `window.__pendingQuickAddFlowSource` carries a * suggested Flow Source value (e.g. from Bills → Pay) into the next * Quick Add save. The Quick Add form has no visible Flow Source control, so * this is the bridge that keeps Cash Flow Flow Source in sync with the * originating bill's Payment Source without redesigning the UI. * * Parallel channel: `window.__pendingUpcomingPaymentId` carries an Upcoming * row ID when Quick Add was launched from the Upcoming board. savePayment() * consumes it on a successful save and calls applyPaymentToUpcomingExpense() * so the Upcoming row's remaining balance is decremented (and auto-flipped * to Paid when it hits 0). Cleared on cold tab load or after a save. * * Lifecycle: * - prefillQuickPayment() sets it when the caller supplies a flowSource. * - savePayment() reads it, sends it to quickAddPayment(), and clears it * on success so a later manual save doesn't inherit a stale value. * - loadPaymentSection() clears it whenever the tab is loaded cold. */ function prefillQuickPayment(prefill) { if (!prefill) return; showPage('cashflow'); showTab('payments'); if (prefill.entryType) { document.getElementById('pay_type').value = prefill.entryType; } if (prefill.entryDate) { document.getElementById('pay_date').value = prefill.entryDate; } filterPaymentPayees(); const input = document.getElementById('pay_payeeInput'); const select = document.getElementById('pay_payeeSelect'); input.value = ''; select.value = ''; if (prefill.payee) { const wantedRaw = String(prefill.payee || '').trim(); const wanted = wantedRaw.toLowerCase(); let matchedValue = ''; for (let i = 0; i < select.options.length; i++) { const optValue = String(select.options[i].value || '').trim(); if (optValue.toLowerCase() === wanted) { matchedValue = optValue; break; } } if (matchedValue) { select.value = matchedValue; input.value = matchedValue; } else { input.value = wantedRaw; } // Async-data race fix: when Quick Add hasn't been visited yet this // session, paymentPayees is still [] at the moment prefill runs // (showTab('payments') just fired loadPaymentSection() and its RPC // hasn't returned). The match loop above runs against an empty // dropdown and falls through to "input only", so the user sees // the typed-text payee filled in but the Existing Payee dropdown // stays blank. We stash the wanted payee here; loadPaymentSection's // success handler reapplies the match once the dropdown is // populated. Cleared on consume so a later manual save / cold tab // load can't inherit a stale prefill. Same lifecycle pattern as // window.__pendingQuickAddFlowSource / __pendingUpcomingPaymentId. window.__pendingQuickAddPayee = wantedRaw; } if (prefill.amount !== null && prefill.amount !== undefined) { setCurrencyInputValue('pay_amount', prefill.amount); } else { document.getElementById('pay_amount').value = ''; } // Normalize and stash the optional flowSource so the next savePayment() // call can include it in the quickAddPayment request. Any value other // than CASH/CREDIT_CARD is ignored — backend validation would reject it // anyway and we don't want a bad value lingering across saves. var rawFlowSource = String(prefill.flowSource || '').trim().toUpperCase().replace(/[\s-]+/g, '_'); if (rawFlowSource === 'CASH' || rawFlowSource === 'CREDIT_CARD') { window.__pendingQuickAddFlowSource = rawFlowSource; } else { window.__pendingQuickAddFlowSource = ''; } // Upcoming → Quick Add bridge: stash the originating Upcoming row id so // savePayment()'s success handler can decrement the remaining balance and // close out the row when fully paid. Non-Upcoming prefills clear it so a // stray earlier value can't follow a later Quick Add save. window.__pendingUpcomingPaymentId = String(prefill.upcomingId || '').trim(); loadPaymentPreview(); } function loadPaymentSection() { document.getElementById('pay_date').value = new Date().toISOString().slice(0, 10); // Any pending prefill-only Flow Source value only makes sense inside the // single save that immediately follows a prefill. Clear it on a cold // tab load so a later manual Quick Add never inherits it. window.__pendingQuickAddFlowSource = ''; window.__pendingUpcomingPaymentId = ''; // NOTE: do NOT clear window.__pendingQuickAddPayee here. // prefillQuickPayment() sets that variable AFTER calling // showTab('payments') (which triggers this function), so an entry-time // clear would race-erase a freshly-set prefill. The success handler // below consumes-and-clears it once the dropdown is populated. google.script.run .withSuccessHandler(function(data) { paymentPayees = data.payees || []; filterPaymentPayees(); consumePendingQuickAddPayeePrefill_(); loadPaymentPreview(); }) .withFailureHandler(function(err) { setStatus('pay_status', (err.message || String(err)), true); }) .getQuickAddPaymentUiData(); } /** * Reapplies a pending payee prefill (set by prefillQuickPayment) once * the Existing Payee dropdown has been freshly populated. Without this, * launching Quick Add from Bills → Pay on a cold tab leaves the * dropdown blank because the synchronous match attempt in * prefillQuickPayment ran before paymentPayees finished loading. * * Guards: * - Only re-applies if the typed-text input still equals the * originally-wanted payee (i.e. the user hasn't started editing it * in the brief window between prefill and data load). This avoids * clobbering a manual edit if the data lands very late. * - Always clears window.__pendingQuickAddPayee after consume so a * later manual save / future prefill can't inherit a stale value. */ function consumePendingQuickAddPayeePrefill_() { const wantedRaw = String(window.__pendingQuickAddPayee || '').trim(); if (!wantedRaw) return; window.__pendingQuickAddPayee = ''; const select = document.getElementById('pay_payeeSelect'); const input = document.getElementById('pay_payeeInput'); if (!select || !input) return; // Don't override a user edit that happened during the async window. if (String(input.value || '').trim() !== wantedRaw) return; const wanted = wantedRaw.toLowerCase(); for (let i = 0; i < select.options.length; i++) { const optValue = String(select.options[i].value || '').trim(); if (optValue.toLowerCase() === wanted) { select.value = optValue; input.value = optValue; return; } } } function filterPaymentPayees() { const type = document.getElementById('pay_type').value; const sel = document.getElementById('pay_payeeSelect'); sel.innerHTML = ''; paymentPayees.filter(function(p) { return p.type === type; }).forEach(function(p) { const opt = document.createElement('option'); opt.value = p.payee; opt.textContent = p.payee; sel.appendChild(opt); }); // Always offer an explicit "Other" sentinel so users know they can // confidently add a brand-new payee without wondering whether to // ignore the dropdown. Selecting it clears+focuses the typed input // via syncPaymentPayeeInput; currentPaymentPayee skips the sentinel // so the save path always sees the typed value, not "__OTHER__". const otherOpt = document.createElement('option'); otherOpt.value = QUICK_ADD_PAYEE_OTHER_SENTINEL_; otherOpt.textContent = 'Other (type new payee)'; sel.appendChild(otherOpt); } function loadPaymentPreview() { const payload = { entryType: document.getElementById('pay_type').value, payee: currentPaymentPayee(), entryDate: document.getElementById('pay_date').value }; if (!payload.payee || !payload.entryDate) { document.getElementById('pay_sheetName').textContent = '—'; document.getElementById('pay_monthName').textContent = '—'; document.getElementById('pay_currentValue').textContent = '—'; document.getElementById('pay_rowExists').textContent = '—'; applyQuickPaymentPriorMonth_(null); return; } google.script.run .withSuccessHandler(function(data) { document.getElementById('pay_sheetName').textContent = data.sheetName || '—'; document.getElementById('pay_monthName').textContent = data.month || '—'; document.getElementById('pay_currentValue').textContent = fmtCurrency(data.currentValue); document.getElementById('pay_rowExists').textContent = data.rowExists ? 'Yes' : 'No'; applyQuickPaymentPriorMonth_(data); }) .withFailureHandler(function(err) { setStatus('pay_status', (err.message || String(err)), true); }) .getQuickAddPreview(payload); } function applyQuickAddPreviewFromResult_(data) { if (!data) return; document.getElementById('pay_sheetName').textContent = data.sheetName || '—'; document.getElementById('pay_monthName').textContent = data.month || '—'; document.getElementById('pay_currentValue').textContent = fmtCurrency(data.currentValue); document.getElementById('pay_rowExists').textContent = data.rowExists ? 'Yes' : 'No'; applyQuickPaymentPriorMonth_(data); } function runDebtPlannerAfterQuickPayment_() { // Background planner refresh — same silent-follow-up pattern Planning → // Debts uses on save (saveDebt() in Dashboard_Script_PlanningDebts.html). // The Quick Add server function (quickAddPayment) deliberately does NOT // run the debt planner inline so the user gets "Saved to Cash Flow." // immediately, in well under a second; that's the same reason // updateDebtField stopped running the planner inline. This call is the // delayed catch-up so Rolling Debt Payoff and planner-derived KPIs still // update shortly after save. // // Two behaviors we picked up from the Debts pattern: // 1) runPlannerAndRefreshDashboard (not bare runDebtPlanner) so the // planner's freshly-rebuilt snapshot is returned and applied — // otherwise the dashboard would keep showing pre-planner data // until the user manually refreshed. // 2) Failures are silent. The Cash Flow row + LOG - Activity row are // already saved on the sheet at this point and the user has seen // "Saved to Cash Flow.", so a planner glitch must not overwrite // that success message with a red error in `pay_status`. Persistent // planner failure surfaces clearly via the top-bar Run Planner Now // button, which is the explicit re-run path. try { google.script.run .withSuccessHandler(function(plannerRes) { if (plannerRes && plannerRes.snapshot && typeof applySnapshot === 'function') { applySnapshot(plannerRes.snapshot); } }) .withFailureHandler(function(_err) { /* silent: see comment above */ }) .runPlannerAndRefreshDashboardFromSave(); } catch (_e) { /* silent */ } } function savePayment() { const rawAmount = document.getElementById('pay_amount').value; const numericAmount = toNumber(rawAmount); if (String(rawAmount || '').trim() && isNaN(numericAmount)) { setStatus('pay_status', 'Amount must be a valid number.', true); return; } // Pull the prefill-only Flow Source hint (set by prefillQuickPayment() for // flows like Bills → Pay). Consumed on this one save so a subsequent // manual Quick Add never silently reuses the previous bill's source. var pendingFlowSource = window.__pendingQuickAddFlowSource || ''; // Upcoming → Quick Add: capture then immediately clear the pending id so // a late retry of this handler can't apply the payment twice. var pendingUpcomingId = window.__pendingUpcomingPaymentId || ''; window.__pendingUpcomingPaymentId = ''; // Show plain-text "Saving…" feedback the user will always see even if a // stale deploy or style override hides the richer spinner markup. The // success handler below replaces this with "Saved to Cash Flow." — same // pattern the legacy sidebar savePayment() has used reliably for ages. // Layer the spinner version on top so browsers/deploys with the newer // dash-loading CSS still get the animated indicator. setStatus('pay_status', 'Saving…', false); if (typeof setStatusLoading === 'function') { try { setStatusLoading('pay_status', 'Saving…'); } catch (_e) {} } google.script.run .withSuccessHandler(function(res) { setStatus('pay_status', res.message || 'Saved.', false); if (res && res.preview) { applyQuickAddPreviewFromResult_(res.preview); } else { loadPaymentPreview(); } document.getElementById('pay_amount').value = ''; window.__pendingQuickAddFlowSource = ''; // Activity ledger refresh — two-step optimistic update so the new // row appears INSTANTLY (no RPC) and survives the background // server reconcile. // // Step 1: prepend the row directly from `res.activitySnapshot`, // which `quickAddPayment` already returned with every field we // need (entry type, payee, amount, dates, sheet, month label). // The user sees the new row at the top of the Activity table // before the next animation frame — even on slow networks. // // Step 2: fire `loadActivitySection({ quiet: true })` to reconcile // with the authoritative server list. `quiet: true` skips the // "Loading activity…" placeholder so the optimistic row stays // visible during the reconcile (which can take seconds when the // user just kicked off 4 other RPCs including the planner). When // the server returns, `window.__activityRows` is replaced with // the real list (which contains the same event) and the table // re-renders. Net effect: row visible instantly, no flicker. // // Both calls use typeof guards because the Activity script lives // in a separate HTML include — if it hasn't loaded yet the Quick // Add save still completes cleanly. if (typeof prependOptimisticQuickPayActivityRow_ === 'function') { try { prependOptimisticQuickPayActivityRow_(res && res.activitySnapshot); } catch (_e) { /* non-fatal */ } } if (typeof loadActivitySection === 'function') { try { loadActivitySection({ quiet: true }); } catch (_e) { /* non-fatal */ } } // Chain the Upcoming balance adjustment after the Quick Add row is // safely written, then let the Upcoming / action-section refreshes // run once at the end so we don't double-render. if (pendingUpcomingId) { google.script.run .withSuccessHandler(function(upRes) { if (upRes && upRes.message) { setStatus('pay_status', (res && res.message ? res.message + ' ' : '') + upRes.message, false); } refreshSnapshot(); loadUpcomingSection(); loadDashboardActionSections(); runDebtPlannerAfterQuickPayment_(); }) .withFailureHandler(function(err) { // The Cash Flow row is already saved at this point. Surface the // follow-up error but still refresh the board so the user sees // the current state. setStatus('pay_status', 'Payment saved, but Upcoming update failed: ' + (err.message || String(err)), true); refreshSnapshot(); loadUpcomingSection(); loadDashboardActionSections(); runDebtPlannerAfterQuickPayment_(); }) .applyPaymentToUpcomingExpense(pendingUpcomingId, numericAmount); } else { refreshSnapshot(); loadUpcomingSection(); loadDashboardActionSections(); runDebtPlannerAfterQuickPayment_(); } }) .withFailureHandler(function(err) { setStatus('pay_status', (err.message || String(err)), true); }) .quickAddPayment({ entryType: document.getElementById('pay_type').value, payee: currentPaymentPayee(), entryDate: document.getElementById('pay_date').value, amount: numericAmount, createIfMissing: document.getElementById('pay_createIfMissing').checked, flowSource: pendingFlowSource }); }
