function currentPaymentPayee() { const typed = document.getElementById('pay_payeeInput').value.trim(); if (typed) return typed; return document.getElementById('pay_payeeSelect').value || ''; } function syncPaymentPayeeInput() { const selected = document.getElementById('pay_payeeSelect').value || ''; document.getElementById('pay_payeeInput').value = selected; } function applyQuickPaymentPriorMonth_(data) { const labelEl = document.getElementById('pay_priorMonthLabel'); const valEl = document.getElementById('pay_priorMonthPayment'); if (!valEl) return; if (labelEl) labelEl.textContent = data && data.priorMonthLabel ? data.priorMonthLabel : '—'; if (!data) { valEl.textContent = '—'; return; } if (data.priorMonthUnavailableMessage) { valEl.textContent = data.priorMonthUnavailableMessage; return; } if (data.priorMonthValue !== null && data.priorMonthValue !== undefined && data.priorMonthValue !== '') { valEl.textContent = fmtCurrency(data.priorMonthValue); } else { valEl.textContent = '—'; } } /** * Hidden prefill-only state: `window.__pendingQuickAddFlowSource` carries a * suggested Flow Source value (e.g. from Bills → Pay) into the next * Quick Add save. The Quick Add form has no visible Flow Source control, so * this is the bridge that keeps Cash Flow Flow Source in sync with the * originating bill's Payment Source without redesigning the UI. * * Lifecycle: * - prefillQuickPayment() sets it when the caller supplies a flowSource. * - savePayment() reads it, sends it to quickAddPayment(), and clears it * on success so a later manual save doesn't inherit a stale value. * - loadPaymentSection() clears it whenever the tab is loaded cold. */ function prefillQuickPayment(prefill) { if (!prefill) return; showPage('cashflow'); showTab('payments'); if (prefill.entryType) { document.getElementById('pay_type').value = prefill.entryType; } if (prefill.entryDate) { document.getElementById('pay_date').value = prefill.entryDate; } filterPaymentPayees(); const input = document.getElementById('pay_payeeInput'); const select = document.getElementById('pay_payeeSelect'); input.value = ''; select.value = ''; if (prefill.payee) { const wantedRaw = String(prefill.payee || '').trim(); const wanted = wantedRaw.toLowerCase(); let matchedValue = ''; for (let i = 0; i < select.options.length; i++) { const optValue = String(select.options[i].value || '').trim(); if (optValue.toLowerCase() === wanted) { matchedValue = optValue; break; } } if (matchedValue) { select.value = matchedValue; input.value = matchedValue; } else { input.value = wantedRaw; } } if (prefill.amount !== null && prefill.amount !== undefined) { setCurrencyInputValue('pay_amount', prefill.amount); } else { document.getElementById('pay_amount').value = ''; } // Normalize and stash the optional flowSource so the next savePayment() // call can include it in the quickAddPayment request. Any value other // than CASH/CREDIT_CARD is ignored — backend validation would reject it // anyway and we don't want a bad value lingering across saves. var rawFlowSource = String(prefill.flowSource || '').trim().toUpperCase().replace(/[\s-]+/g, '_'); if (rawFlowSource === 'CASH' || rawFlowSource === 'CREDIT_CARD') { window.__pendingQuickAddFlowSource = rawFlowSource; } else { window.__pendingQuickAddFlowSource = ''; } loadPaymentPreview(); } function loadPaymentSection() { document.getElementById('pay_date').value = new Date().toISOString().slice(0, 10); // Any pending prefill-only Flow Source value only makes sense inside the // single save that immediately follows a prefill. Clear it on a cold // tab load so a later manual Quick Add never inherits it. window.__pendingQuickAddFlowSource = ''; google.script.run .withSuccessHandler(function(data) { paymentPayees = data.payees || []; filterPaymentPayees(); loadPaymentPreview(); }) .withFailureHandler(function(err) { setStatus('pay_status', 'Error: ' + (err.message || String(err)), true); }) .getQuickAddPaymentUiData(); } function filterPaymentPayees() { const type = document.getElementById('pay_type').value; const sel = document.getElementById('pay_payeeSelect'); sel.innerHTML = ''; paymentPayees.filter(function(p) { return p.type === type; }).forEach(function(p) { const opt = document.createElement('option'); opt.value = p.payee; opt.textContent = p.payee; sel.appendChild(opt); }); } function loadPaymentPreview() { const payload = { entryType: document.getElementById('pay_type').value, payee: currentPaymentPayee(), entryDate: document.getElementById('pay_date').value }; if (!payload.payee || !payload.entryDate) { document.getElementById('pay_sheetName').textContent = '—'; document.getElementById('pay_monthName').textContent = '—'; document.getElementById('pay_currentValue').textContent = '—'; document.getElementById('pay_rowExists').textContent = '—'; applyQuickPaymentPriorMonth_(null); return; } google.script.run .withSuccessHandler(function(data) { document.getElementById('pay_sheetName').textContent = data.sheetName || '—'; document.getElementById('pay_monthName').textContent = data.month || '—'; document.getElementById('pay_currentValue').textContent = fmtCurrency(data.currentValue); document.getElementById('pay_rowExists').textContent = data.rowExists ? 'Yes' : 'No'; applyQuickPaymentPriorMonth_(data); }) .withFailureHandler(function(err) { setStatus('pay_status', 'Error: ' + (err.message || String(err)), true); }) .getQuickAddPreview(payload); } function applyQuickAddPreviewFromResult_(data) { if (!data) return; document.getElementById('pay_sheetName').textContent = data.sheetName || '—'; document.getElementById('pay_monthName').textContent = data.month || '—'; document.getElementById('pay_currentValue').textContent = fmtCurrency(data.currentValue); document.getElementById('pay_rowExists').textContent = data.rowExists ? 'Yes' : 'No'; applyQuickPaymentPriorMonth_(data); } function runDebtPlannerAfterQuickPayment_() { google.script.run .withFailureHandler(function(err) { setStatus('pay_status', 'Planner error: ' + (err.message || String(err)), true); }) .runDebtPlanner(); } function savePayment() { const rawAmount = document.getElementById('pay_amount').value; const numericAmount = toNumber(rawAmount); if (String(rawAmount || '').trim() && isNaN(numericAmount)) { setStatus('pay_status', 'Error: Amount must be a valid number.', true); return; } // Pull the prefill-only Flow Source hint (set by prefillQuickPayment() for // flows like Bills → Pay). Consumed on this one save so a subsequent // manual Quick Add never silently reuses the previous bill's source. var pendingFlowSource = window.__pendingQuickAddFlowSource || ''; google.script.run .withSuccessHandler(function(res) { setStatus('pay_status', res.message || 'Saved.', false); if (res && res.preview) { applyQuickAddPreviewFromResult_(res.preview); } else { loadPaymentPreview(); } document.getElementById('pay_amount').value = ''; window.__pendingQuickAddFlowSource = ''; refreshSnapshot(); loadUpcomingSection(); loadDashboardActionSections(); runDebtPlannerAfterQuickPayment_(); }) .withFailureHandler(function(err) { setStatus('pay_status', 'Error: ' + (err.message || String(err)), true); }) .quickAddPayment({ entryType: document.getElementById('pay_type').value, payee: currentPaymentPayee(), entryDate: document.getElementById('pay_date').value, amount: numericAmount, createIfMissing: document.getElementById('pay_createIfMissing').checked, flowSource: pendingFlowSource }); }
