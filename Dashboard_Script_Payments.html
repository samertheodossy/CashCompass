Any value other // than CASH/CREDIT_CARD is ignored — backend validation would reject it // anyway and we don't want a bad value lingering across saves. var rawFlowSource = String(prefill.flowSource || '').trim().toUpperCase().replace(/[\s-]+/g, '_'); if (rawFlowSource === 'CASH' || rawFlowSource === 'CREDIT_CARD') { window.__pendingQuickAddFlowSource = rawFlowSource; } else { window.__pendingQuickAddFlowSource = ''; } // Upcoming → Quick Add bridge: stash the originating Upcoming row id so // savePayment()'s success handler can decrement the remaining balance and // close out the row when fully paid. Non-Upcoming prefills clear it so a // stray earlier value can't follow a later Quick Add save. window.__pendingUpcomingPaymentId = String(prefill.upcomingId || '').trim(); loadPaymentPreview(); } function loadPaymentSection() { document.getElementById('pay_date').value = new Date().toISOString().slice(0, 10); // Any pending prefill-only Flow Source value only makes sense inside the // single save that immediately follows a prefill. Clear it on a cold // tab load so a later manual Quick Add never inherits it. window.__pendingQuickAddFlowSource = ''; window.__pendingUpcomingPaymentId = ''; google.script.run .withSuccessHandler(function(data) { paymentPayees = data.payees || []; filterPaymentPayees(); loadPaymentPreview(); }) .withFailureHandler(function(err) { setStatus('pay_status', (err.message || String(err)), true); }) .getQuickAddPaymentUiData(); } function filterPaymentPayees() { const type = document.getElementById('pay_type').value; const sel = document.getElementById('pay_payeeSelect'); sel.innerHTML = ''; paymentPayees.filter(function(p) { return p.type === type; }).forEach(function(p) { const opt = document.createElement('option'); opt.value = p.payee; opt.textContent = p.payee; sel.appendChild(opt); }); } function loadPaymentPreview() { const payload = { entryType: document.getElementById('pay_type').value, payee: currentPaymentPayee(), entryDate: document.getElementById('pay_date').value }; if (!payload.payee || !payload.entryDate) { document.getElementById('pay_sheetName').textContent = '—'; document.getElementById('pay_monthName').textContent = '—'; document.getElementById('pay_currentValue').textContent = '—'; document.getElementById('pay_rowExists').textContent = '—'; applyQuickPaymentPriorMonth_(null); return; } google.script.run .withSuccessHandler(function(data) { document.getElementById('pay_sheetName').textContent = data.sheetName || '—'; document.getElementById('pay_monthName').textContent = data.month || '—'; document.getElementById('pay_currentValue').textContent = fmtCurrency(data.currentValue); document.getElementById('pay_rowExists').textContent = data.rowExists ? 'Yes' : 'No'; applyQuickPaymentPriorMonth_(data); }) .withFailureHandler(function(err) { setStatus('pay_status', (err.message || String(err)), true); }) .getQuickAddPreview(payload); } function applyQuickAddPreviewFromResult_(data) { if (!data) return; document.getElementById('pay_sheetName').textContent = data.sheetName || '—'; document.getElementById('pay_monthName').textContent = data.month || '—'; document.getElementById('pay_currentValue').textContent = fmtCurrency(data.currentValue); document.getElementById('pay_rowExists').textContent = data.rowExists ? 'Yes' : 'No'; applyQuickPaymentPriorMonth_(data); } function runDebtPlannerAfterQuickPayment_() { google.script.run .withFailureHandler(function(err) { setStatus('pay_status', (err.message || String(err)), true); }) .runDebtPlanner(); } function savePayment() { const rawAmount = document.getElementById('pay_amount').value; const numericAmount = toNumber(rawAmount); if (String(rawAmount || '').trim() && isNaN(numericAmount)) { setStatus('pay_status', 'Amount must be a valid number.', true); return; } // Pull the prefill-only Flow Source hint (set by prefillQuickPayment() for // flows like Bills → Pay). Consumed on this one save so a subsequent // manual Quick Add never silently reuses the previous bill's source. var pendingFlowSource = window.__pendingQuickAddFlowSource || ''; // Upcoming → Quick Add: capture then immediately clear the pending id so // a late retry of this handler can't apply the payment twice. var pendingUpcomingId = window.__pendingUpcomingPaymentId || ''; window.__pendingUpcomingPaymentId = ''; google.script.run .withSuccessHandler(function(res) { setStatus('pay_status', res.message || 'Saved.', false); if (res && res.preview) { applyQuickAddPreviewFromResult_(res.preview); } else { loadPaymentPreview(); } document.getElementById('pay_amount').value = ''; window.__pendingQuickAddFlowSource = ''; // Chain the Upcoming balance adjustment after the Quick Add row is // safely written, then let the Upcoming / action-section refreshes // run once at the end so we don't double-render. if (pendingUpcomingId) { google.script.run .withSuccessHandler(function(upRes) { if (upRes && upRes.message) { setStatus('pay_status', (res && res.message ? res.message + ' ' : '') + upRes.message, false); } refreshSnapshot(); loadUpcomingSection(); loadDashboardActionSections(); runDebtPlannerAfterQuickPayment_(); }) .withFailureHandler(function(err) { // The Cash Flow row is already saved at this point. Surface the // follow-up error but still refresh the board so the user sees // the current state. setStatus('pay_status', 'Payment saved, but Upcoming update failed: ' + (err.message || String(err)), true); refreshSnapshot(); loadUpcomingSection(); loadDashboardActionSections(); runDebtPlannerAfterQuickPayment_(); }) .applyPaymentToUpcomingExpense(pendingUpcomingId, numericAmount); } else { refreshSnapshot(); loadUpcomingSection(); loadDashboardActionSections(); runDebtPlannerAfterQuickPayment_(); } }) .withFailureHandler(function(err) { setStatus('pay_status', (err.message || String(err)), true); }) .quickAddPayment({ entryType: document.getElementById('pay_type').value, payee: currentPaymentPayee(), entryDate: document.getElementById('pay_date').value, amount: numericAmount, createIfMissing: document.getElementById('pay_createIfMissing').checked, flowSource: pendingFlowSource }); }
