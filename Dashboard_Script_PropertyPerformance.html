function loadPropertyPerformanceSection() { const yearSel = document.getElementById('pp_year'); const year = yearSel && yearSel.value ? Number(yearSel.value) : new Date().getFullYear(); setStatusLoading('pp_status'); const msgEl = document.getElementById('pp_message'); if (msgEl) msgEl.textContent = ''; const tbodyLoading = document.getElementById('pp_tbody'); if (tbodyLoading) { tbodyLoading.innerHTML = '' + loadingBlockHtml('Loading…', true) + ''; } google.script.run .withSuccessHandler(function(data) { setStatus('pp_status', '', false); if (data && data.yearOptions && yearSel) { const currentVal = String(data.year); yearSel.innerHTML = ''; (data.yearOptions || []).forEach(function(y) { const opt = document.createElement('option'); opt.value = String(y); opt.textContent = String(y); if (String(y) === currentVal) opt.selected = true; yearSel.appendChild(opt); }); } if (msgEl && data && data.message) { msgEl.textContent = data.message; } const p = (data && data.portfolio) ? data.portfolio : {}; const setMini = function(id, v) { const el = document.getElementById(id); if (!el) return; el.textContent = fmtCurrency(v); el.classList.toggle('currency-negative', Number(v) < 0); }; setMini('pp_port_equity', p.equity); setMini('pp_port_rent', p.rent); setMini('pp_port_exp', p.operatingExpenses != null ? p.operatingExpenses : p.expenses); setMini('pp_port_loan_payments', p.loanPayments); setMini('pp_port_net_cash_flow', p.netCashFlow); const tbody = document.getElementById('pp_tbody'); if (!tbody) return; const rows = (data && data.rows) ? data.rows : []; if (!rows.length) { tbody.innerHTML = 'No houses yet.'; return; } tbody.innerHTML = rows.map(function(row) { const ptype = row.propertyType ? escapeHtml(row.propertyType) : '—'; const amountClass = function(value) { return Number(value) < 0 ? ' class="currency-negative"' : ''; }; const financingNote = row.financingMessage ? '
' + escapeHtml(row.financingMessage) + '
' : ''; return ( '' + '' + escapeHtml(row.house) + '' + '' + ptype + '' + '' + fmtCurrency(row.currentValue) + '' + '' + fmtCurrency(row.loanAmount) + '' + '' + fmtCurrency(row.equity) + '' + '' + fmtCurrency(row.rent) + '' + '' + fmtCurrency(row.operatingExpenses) + '' + '' + fmtCurrency(row.loanPayments) + financingNote + '' + '' + fmtCurrency(row.netCashFlow) + '' + '' ); }).join(''); }) .withFailureHandler(function(err) { setStatus('pp_status', (err.message || String(err)), true); const tbody = document.getElementById('pp_tbody'); if (tbody) { tbody.innerHTML = 'Could not load.'; } }) .getPropertyPerformanceData({ year: year }); }
