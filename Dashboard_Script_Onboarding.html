In test mode // the server-side resolver routes every read/write to TEST - * sheets. // -------------------------------------------------------------------------- function onboardingMode() { try { if (typeof window !== 'undefined' && window.location && window.location.search) { if (/[?&]onboarding=test\b/i.test(window.location.search)) return 'test'; } } catch (_e) { /* ignore */ } return 'normal'; } // View switching inside Setup. Setup currently has three screens: // 'welcome' — introduction + Continue/Back buttons // 'grid' — status review cards (overview of every Setup area) // 'bank' — Bank Accounts detail (first step-card drill-down) // Simple DOM show/hide keeps this localized. Adding more detail views // later (Debts/Bills/Upcoming/Income) reuses this switch. function onboardingShowView(view) { var views = { welcome: document.getElementById('onboarding_view_welcome'), grid: document.getElementById('onboarding_view_grid'), bank: document.getElementById('onboarding_view_bank') }; // Guard: at minimum Welcome + Grid must exist. ' ' + cls : ''); } // Map internal Complete/Partial/Missing state to user-facing wording. // Display-only: the server still returns the three granular states so // we can style/route on them, but the UI collapses Partial+Missing // into a single "Not set up" label for now to keep the flow simple. function onboardingStatusLabel_(status) { if (status === 'complete') return 'Setup complete'; return 'Not set up'; } function onboardingRenderStepCard_(step) { var badge = document.querySelector('[data-badge="' + step.key + '"]'); var body = document.querySelector('[data-body="' + step.key + '"]'); if (!badge || !body) return; var statusClass = 'status-' + (step.status || 'missing'); badge.className = 'onboarding-step-badge ' + statusClass; badge.textContent = onboardingStatusLabel_(step.status); var fragments = []; if (step.note) { fragments.push('

'' : 'none'; } if (createBtn) { createBtn.style.display = (mode === 'test') ? '' : 'none'; createBtn.disabled = false; createBtn.textContent = 'Create missing TEST sheets'; } // Always start at Welcome whenever the user (re)enters Setup. This // keeps the flow predictable and makes the status grid an explicit // step rather than an automatic page-view. onboardingShowView('welcome'); } // Internal: fetch and render the status grid. Triggered by the Continue // button on the Welcome screen or the Refresh button on the grid itself. function onboardingLoadStatusGrid_() { var mode = onboardingMode(); var banner = document.getElementById('onboarding_test_banner'); var createBtn = document.getElementById('onboarding_create_test_sheets_btn'); onboardingSetStatus_('Reading status…', 'dash-status-loading'); google.script.run .withSuccessHandler(function(data) { if (!data || !Array.isArray(data.steps)) { onboardingSetStatus_('Could not read setup status.', 'error'); return; } data.steps.forEach(onboardingRenderStepCard_); if (banner && data.testBanner) banner.textContent = data.testBanner; if (mode === 'test' && createBtn) { if (data.missingTestSheets && data.missingTestSheets.length > 0) { createBtn.style.display = ''; createBtn.textContent = 'Create missing TEST sheets (' + data.missingTestSheets.length + ')'; } else { createBtn.style.display = ''; createBtn.textContent = 'TEST sheets already present'; createBtn.disabled = true; } } var total = data.steps.length; var complete = data.steps.filter(function(s) { return s.status === 'complete'; }).length; var notSetUp = total - complete; // Match the user-facing labels used on each card: "Setup complete" // vs "Not set up". Renders a simple account list; no edit affordances. // -------------------------------------------------------------------------- function onboardingBankSetStatus_(text, cls) { var el = document.getElementById('onboarding_bank_status'); if (!el) return; el.textContent = text || ''; el.className = 'status' + (cls ? ' ' + cls : ''); } function onboardingLoadBankDetail_() { var mode = onboardingMode(); var body = document.getElementById('onboarding_bank_body'); var badge = document.getElementById('onboarding_bank_badge'); if (badge) { badge.className = 'onboarding-detail-badge'; badge.textContent = '…'; } if (body) { body.innerHTML = '

Loading accounts…

'; } onboardingBankSetStatus_('Reading accounts…', 'dash-status-loading'); google.script.run .withSuccessHandler(function(data) { onboardingRenderBankDetail_(data); }) .withFailureHandler(function(err) { onboardingBankSetStatus_( 'Error reading bank accounts: ' + (err && err.message ? err.message : String(err)), 'error' ); if (body) body.innerHTML = ''; }) .getOnboardingBankAccountsFromDashboard(mode); } function onboardingRenderBankDetail_(data) { var body = document.getElementById('onboarding_bank_body'); var badge = document.getElementById('onboarding_bank_badge'); if (!body) return; if (!data) { onboardingBankSetStatus_('Could not read bank accounts.', 'error'); body.innerHTML = ''; return; } var status = data.status || 'missing'; if (badge) { badge.className = 'onboarding-detail-badge status-' + status; badge.textContent = onboardingStatusLabel_(status); } var summary = []; var count = (data.accounts && data.accounts.length) || 0; summary.push(count + (count === 1 ? 'No bank accounts are set up yet.' : 'No bank accounts tracked yet' + (data.year ? ' for ' + escapeOnboardingHtml_(String(data.year)) : '') + '.'; var ctaHtml = onboardingMode() === 'test' ? '

' + rows + '
' ); body.innerHTML = fragments.join(''); } // Jump into the existing Bank Accounts management surface. In normal // mode this is the Assets page → Bank Accounts tab. In test mode there // is no equivalent destination (the main dashboard is not scoped to // TEST sheets), so we keep the user on this read-only view and surface // a status hint instead of silently leaving Setup. function onboardingOpenBankAccountsPage() { if (onboardingMode() === 'test') { onboardingBankSetStatus_( 'Editing is not available in TEST MODE in this pass.', '' ); return; } // Safeguard: make sure INPUT - Bank Accounts exists *before* handing the // user off to the Assets → Bank Accounts editor. If it is missing, the // backend creates it with the minimum valid structure the rest of the // app expects. Navigation only happens after the ensure call succeeds // so the user never lands on a broken editor screen. onboardingBankSetStatus_('Preparing Bank Accounts editor…', 'dash-status-loading'); var navigate = function() { // Open the real editor in Setup mode so the shell (main nav, sub-tabs, // top-right controls) collapses into a focused continuation of Setup. // The 'bank' return-view tells exitSetupEditorMode() which onboarding // detail screen to restore when the user clicks "Back to Setup". enterSetupEditorMode('assets', 'bank', 'bank'); }; google.script.run .withSuccessHandler(function(resp) { if (!resp || resp.ok === false) { var reason = resp && resp.reason ? resp.reason : 'Could not prepare the Bank Accounts sheet.'; onboardingBankSetStatus_(reason, 'error'); return; } onboardingBankSetStatus_('', ''); navigate(); }) .withFailureHandler(function(err) { onboardingBankSetStatus_( 'Could not prepare the Bank Accounts sheet: ' + (err && err.message ? err.message : String(err)), 'error' ); }) .ensureOnboardingBankAccountsSheetFromDashboard('normal'); } // -------------------------------------------------------------------------- // Setup editor handoff helpers // -------------------------------------------------------------------------- // Reusable pattern: from an onboarding detail screen, open an existing // editor page (the real one — no second copy) but have the dashboard // render it in a collapsed, Setup-focused shell. The return path is // provided by a single "Back to Setup" button. // // enterSetupEditorMode(pageName, tabName, returnView) // pageName — the workspace page to show (e.g. 'assets') // tabName — the sub-tab to activate on that page (e.g. 'bank') // returnView — which onboarding view to restore on Back // ('bank' | 'grid' | 'welcome') // // Kept narrow on purpose: future Debts/Bills/Upcoming/Income handoffs // reuse exactly the same entry/exit points, with their own page+tab // arguments.
